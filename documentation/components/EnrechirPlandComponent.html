import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { PlandActionService } from 'src/app/services/pland-action.service';
import { plandActionModel } from 'src/app/models/plandAction';

@Component({
  selector: 'app-enrechir-pland',
  templateUrl: './enrechir-pland.component.html',
  styleUrls: ['./enrechir-pland.component.css']
})
export class EnrechirPlandComponent implements OnInit {

  constructor(private plandAction: PlandActionService, private route: ActivatedRoute, private router: Router) { }

nom :string='khouloud'

select = true;
  selectedListId: string;
  ngOnInit(): void {
   
    this.route.params.subscribe(
      (params: Params) => {
        console.log(params.id);
        if (params.id) {
          this.selectedListId = params.id;
          this.plandAction.getpland(params.id).subscribe((pland: plandActionModel) => {
            this.pland = pland;
            this.lenght=(pland.equipePlan.length-3)
            for (let i = 0; i < 3; i++) {
              this.userTestStatus= Array(
                {  "name": this.pland.equipePlan[0]},
                {  "name": this.pland.equipePlan[1]},
                {  "name": this.pland.equipePlan[2]},
            );
         
              }
               pland.equipePlan.splice(0,3)
               
           
              console.log(pland.equipePlan[0]);
          })
        } else {
          this.pland = undefined;
        }
      
      }
    )

    
    
  
  }

  toggle(){
    this.select=!this.select;
  }
}
